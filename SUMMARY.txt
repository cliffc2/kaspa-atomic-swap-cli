================================================================================
QUICK START
================================================================================

# Generate secret & hash
SECRET=$(head -c 32 /dev/urandom | xxd -p)
HASH=$(echo -n "$SECRET" | sha256sum | cut -d' ' -f1)

# Create HTLC for 0.5 KAS to peer with 288 block timelock
./target/release/kaspa-atomic-swap-cli initiate \
  --amount 500000000 \
  --to kaspa:qpeer \
  --secret-hash "$HASH" \
  --timelock-blocks 288 \
  --from kaspa:qme \
  --network testnet-uxto \
  --json

# Peer claims by revealing secret
./target/release/kaspa-atomic-swap-cli claim \
  --utxo "<txid>:0" \
  --secret "$SECRET" \
  --json
